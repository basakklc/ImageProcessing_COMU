    return jpg_files

def rotate(im_1):
    m,n,k = im_1.shape
    new=np.zeros((n,m,k),dtype='uint8')
    for i in range (m):
        for j in range(n):
            temp = image_1[i, j]
            new[j,i]=temp
    return new
